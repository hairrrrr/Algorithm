{
    vector<int> arr = a;
    int cnt = 0;
    for(int i = 0; i < arr.size(); ++i)
    {
        int _min = arr[i];
        int _pos = i;
        for(int j = i + 1; j < arr.size(); ++j)
        {
            ++cnt;

            {
               _min = arr[j];
               _pos = j;
            }
        }
        if(_pos != i) swap(arr[i], arr[_pos]);
    }
    cout << cnt << endl;
}

    BubbleSort();
